3.3 Faceted Plots

3.3.1 Random Subsample

Code
set.seed(52242) # for reproducibility

random_n <- 12

ids_subset <- sample(
  x = mydata_long$ID,
  size = random_n,
  replace = FALSE
)

mydata_long |>
  filter(ID %in% ids_subset) |>
  ggplot(
    aes(
      x = age,
      y = depression,
      group = ID
    )
  ) +
  geom_point() +
  geom_smooth(
    method = "lm"
  ) +
  coord_cartesian(
    ylim = c(0, max(mydata_long$depression, na.rm = TRUE))) +
  facet_wrap(
    ~ ID) +
  labs(
    x = "Age (Years)",
    y = "Depression Symptoms"
  ) +
  theme_classic()

facetedPlots <- mydata_long |>

    nrow = 3)

num_pages <- ggforce::n_pages(facetedPlots)

for(i in 1:num_pages){
  print(facetedPlots + 
    ggforce::facet_wrap_paginate(
      ~ ID,
      ncol = 4,
      nrow = 3,
      page = i))
}

# Preamble

## Load Libraries

```{r}
library("tidyverse")
library("ggforce")
```

## Load Data

The data for this example were [simulated](simulate-data.qmd):

```{r}
mydata_long <- read.csv("./data/mydata_long.csv")
```
